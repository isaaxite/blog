```js
window.onload = function() {
  console.log("ready");
  var fileEle = document.querySelector("input[name='file']");
  
  fileEle && fileEle
  .addEventListener("change", function() {
    console.log("event invoke");
    var fileEle = document.querySelector("input[name='file']");
    var windowURL = window.URL || window.webkitURL;
    var imgEle = document.querySelector("#preview");
    var path = windowURL.createObjectURL(fileEle.files[0]);
    
    imgEle.src = path;
    
  }, false);
};
```
